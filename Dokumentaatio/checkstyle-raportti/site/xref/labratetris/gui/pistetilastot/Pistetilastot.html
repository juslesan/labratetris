24   *

29  public class Pistetilastot implements Runnable {
30  
31      private JFrame frame;
32  
33      public Pistetilastot() {
34  
35      }
36  
37      @Override
38      public void run() {
39          frame = new JFrame("Labratetris");
40          frame.setLayout(new BorderLayout());
41  
42          frame.setPreferredSize(new Dimension(350, 600));
43  
44          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
45  
46          try {
47              luoKomponentit(frame.getContentPane());
48          } catch (FileNotFoundException ex) {
49          }
50  
51          frame.pack();
52          frame.setVisible(true);
53  
54      }
55  
56      private void luoKomponentit(Container container) throws FileNotFoundException {
57          JPanel listaus = new JPanel(new GridLayout(6, 1));
58          String[] pistetilastot = pistetilasto();
59  
60          JLabel otsikko = new JLabel("Ennätyspisteet:");
61          listaus.add(otsikko);
62          JLabel ensimmainen = new JLabel(pistetilastot[0]);
63          listaus.add(ensimmainen);
64          JLabel toinen = new JLabel(pistetilastot[1]);
65          listaus.add(toinen);
66          JLabel kolmas = new JLabel(pistetilastot[2]);
67          listaus.add(kolmas);
68          JLabel neljas = new JLabel(pistetilastot[3]);
69          listaus.add(neljas);
70          JLabel viides = new JLabel(pistetilastot[4]);
71          listaus.add(viides);
72  
73          JPanel paneeli = new JPanel(new GridLayout(1, 2));
74          JButton takaisin = new JButton("Takaisin");
75          paneeli.add(takaisin);
76          JButton nollaa = new JButton("Nollaa");
77          paneeli.add(nollaa);
78  
79          container.add(listaus);
80          container.add(paneeli, BorderLayout.PAGE_END);
81          PistetilastotPaneelinKuuntelija kuuntelija = new PistetilastotPaneelinKuuntelija(frame, takaisin, nollaa);
82          takaisin.addActionListener(kuuntelija);
83          nollaa.addActionListener(kuuntelija);
84      }
85  
86      public String[] pistetilasto() throws FileNotFoundException {
87          PisteidenLukija lukija = new PisteidenLukija();
88          return lukija.getPistetilastot();
89      }
90  }
