29  /**
30   *
31   * @author Santeri Juslenius
32   *
33   * Varsinaisen pelin graafisen käyttöliittymän kokonaisuudessaan kasaava luokka.
34   */
35  public class PelinKali implements Runnable {
36  
37      private JFrame frame;
38      private Peli peli;
39      private int sivunPituus;
40      private Tetriskentta tetriskentta;
41      private NaytaVaihtoPalikka vaihtoPalikka;
42      private NaytaSeuraavaPalikka1 seuraavaPalikka1;
43      private NaytaSeuraavaPalikka2 seuraavaPalikka2;
44      private NaytaSeuraavaPalikka3 seuraavaPalikka3;
45      private Infopaneeli infopaneeli;
46  
47      public PelinKali(Peli peli, int sivunPituus) {
48          this.peli = peli;
49          this.sivunPituus = sivunPituus;
50  
51      }
52  
53      @Override
54      public void run() {
55          frame = new JFrame("Labratetris");
56          frame.setLayout(new BorderLayout());
57          int leveys = (peli.getLeveys() * sivunPituus) + sivunPituus * 9;
58          int korkeus = (peli.getKorkeus() * sivunPituus) + sivunPituus * 2 + 30;
59  
60  //        frame.setLayout(new GridLayout(1, 1));
61          frame.setPreferredSize(new Dimension(leveys, korkeus));
62  
63          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
64  
65          luoKomponentit(frame.getContentPane());
66          this.tetriskentta.paivita();
67  
68          paivitaPalikkakentat();
69          paivitaInfopaneeli();
70  
71          lisaaPaivitettavat();
72  
73          frame.pack();
74          frame.setVisible(true);
75          this.peli.setFrame(frame);
76      }
77  
78      /**
79       * Metodi luo kaikki pelin graafisen käyttöliittymän komponentit
80       *
81       * @param container JFramen contentpane
82       */
83      public void luoKomponentit(Container container) {
84          this.tetriskentta = new Tetriskentta(this.peli, this.sivunPituus);
85          frame.addKeyListener(new Nappaimistonkuuntelija(this.peli));
86          JPanel palikkapaneeli = new JPanel();
87          palikkapaneeli.setLayout(new BoxLayout(palikkapaneeli, BoxLayout.PAGE_AXIS));
88  
89          JLabel seuraavat = new JLabel("Seuraavat Palikat:");
90          palikkapaneeli.add(seuraavat);
91  
92          this.seuraavaPalikka1 = new NaytaSeuraavaPalikka1(this.peli, this.sivunPituus);
93          palikkapaneeli.add(seuraavaPalikka1);
94  
95          this.seuraavaPalikka2 = new NaytaSeuraavaPalikka2(this.peli, this.sivunPituus);
96          palikkapaneeli.add(seuraavaPalikka2);
97  
98          this.seuraavaPalikka3 = new NaytaSeuraavaPalikka3(this.peli, this.sivunPituus);
99          palikkapaneeli.add(seuraavaPalikka3);
100 //
101         JLabel vaihto = new JLabel("Vaihto:");
102         palikkapaneeli.add(vaihto);
103 
104         this.vaihtoPalikka = new NaytaVaihtoPalikka(this.peli, this.sivunPituus);
105         palikkapaneeli.add(this.vaihtoPalikka);
106 
107         this.infopaneeli = new Infopaneeli(this.peli);
108 
109 //        container.add(panel, BorderLayout.EAST);
110         container.add(tetriskentta, BorderLayout.CENTER);
111         container.add(infopaneeli, BorderLayout.SOUTH);
112         container.add(palikkapaneeli, BorderLayout.LINE_END);
113 //        container.add(seuraavaPalikka2);
114 
115 //        container.add(vaihtoPalikka, BorderLayout.EAST);
116     }

162      * Lisää Peli luokalle päivitettävät komponentit
163      */
164     public void lisaaPaivitettavat() {
165         this.peli.lisaaPaivitettava(tetriskentta);
166         this.peli.lisaaPaivitettava(seuraavaPalikka1);
167         this.peli.lisaaPaivitettava(seuraavaPalikka2);
168         this.peli.lisaaPaivitettava(seuraavaPalikka3);
169         this.peli.lisaaPaivitettava(infopaneeli);
170         this.peli.lisaaPaivitettava(this.vaihtoPalikka);
